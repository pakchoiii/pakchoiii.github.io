快速排序

void quick_sort(int q[], int l, int r) {
   if (l >= r) return;
 
   int x = q[l + r >> 1], i = l - 1, j = r + 1;
   while (i < j) {
       do i++; while (q[i] < x);
       do j--; while (q[j] > x);
       if (i < j) swap(q[i], q[j]);
   }
 
   quick_sort(q, l, j);       //quick_sort(q,l,i-1); 
                              //用i要保证不用到q的左边界，x取值那也要调整。
   quick_sort(q, j + 1, r);   //quick_sort(q,i,r);
}

void merge_sort(int q[], int l, int r) {
   if (l >= r)return;
   int mid = l + r >> 1;
   merge_sort(q, l, mid), merge_sort(q, mid + 1, r);
   int k = 0, i = l, j = mid + 1;
   while (i <= mid && j <= r) {
       if (q[i] <= q[j])tmp[k++] = q[i++];
       else tmp[k++] = q[j++];
   }
   while (i <= mid)tmp[k++] = q[i++];
   while (j <= r)tmp[k++] = q[j++];
   for (i = l, j = 0; i <= r; i++, j++) {
       q[i] = tmp[j];
   }
}

// 区间[l, r]被划分成[l, mid]和[mid + 1, r]时使用：
intbsearch_1(int l, int r)
{
   while (l < r)
   {
       int mid = l + r >> 1;
       if (check(mid)) r = mid;    // check()判断mid是否满足性质
       else l = mid + 1;
   }
   return l;
}
// 区间[l, r]被划分成[l, mid - 1]和[mid, r]时使用：
intbsearch_2(int l, int r)
{
   while (l < r)
   {
       int mid = l + r + 1 >> 1;
       if (check(mid)) l = mid;
       else r = mid - 1;
   }
   return l;
}

doublebsearch_3(double l, double r)
{
   const double eps = 1e-6;   // eps 表示精度，取决于题目对精度的要求
   while (r - l > eps)
   {
       double mid = (l + r) / 2;
       if (check(mid)) r = mid;
       else l = mid;
   }
   return l;
}

    int t = 0;
    for (int i = 0; i < A.size() || i < B.size(); i++) {
        if (i < A.size())t += A[i];
        if (i < B.size())t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }
    if (t)C.push_back(1);
    return C;
}

class Solution {
public:
    bool cmp(vector<int>& A, vector<int>& B) {
        if (A.size() != B.size())return A.size() > B.size();
        for (int i = A.size()-1; i >= 0; i--) {
            if (A[i] != B[i])return A[i] > B[i];
        }
        return true;
    }
    vector<int> sub(string a, string b)
    {
        vector<int>A, B, C;
        for (int i = a.size() - 1; i >= 0; i--)A.push_back(a[i] - '0'); 
        for (int i = b.size() - 1; i >= 0; i--)B.push_back(b[i] - '0');
        
        if (cmp(A, B)) {
            for (int i = 0, t = 0; i < A.size(); i++)
            {
                t = A[i] - t;
                if (i < B.size()) t -= B[i];
                C.push_back((t + 10) % 10);
                if (t < 0) t = 1;
                else t = 0;
            }
            while (C.size() > 1 && C.back() == 0) C.pop_back();
            reverse(C.begin(), C.end());
        }
        else {
            for (int i = 0, t = 0; i < B.size(); i++)
            {
                t = B[i] - t;
                if (i < A.size()) t -= A[i];
                C.push_back((t + 10) % 10);
                if (t < 0) t = 1;
                else t = 0;
            }

            while (C.size() > 1 && C.back() == 0) C.pop_back();
            reverse(C.begin(), C.end());
            cout << "-";
        }        
        return C;
    }
};
